function largestProduct(digits, consecutive) {
    var r = 0, out = 0,
        start = 0, end = consecutive,
        size = digits.length - consecutive,
        i, d, ld, p, j;

    for (i = 0; i < size; i += 1) {
        d = digits.slice(start + i, end + i);
        ld = d[end - 1];
        if (ld === 0) {
            i += consecutive - 1;
            continue;
        }
        if (ld > out) {
            p = 1;
            for (j = 0; j < consecutive; j += 1) {
                p *= d[j];
            }
            if (p > r) {
                r = p;
            }
        }
        out = d[0];
    }
    
    return r;
}
